22/// This struct contains all basic settings such as number of grid points,
23/// model radius, input and output filenames, etc. Some of these parameters
24/// always need to be set by the user, while others are optional with preset
25/// default values. There is an exception to this rule, namely when restarting
26/// citrus with previously calculated populations. In that case, none of the
27/// non-optional parameters are required.
28#[derive(Deserialize, Debug)]
29#[serde(default, rename_all = "snake_case")]
30pub struct ParameterInput {
31    /// This value sets the outer radius of the computational domain.
32    ///
33    /// ### Note
34    /// It should be set large enough to cover the entire spatial extend of the
35    /// model. In particular, if a cylindrical input model is used (e.g., the
36    /// input file for the RATRAN code) one should not use the radius of the
37    /// cylinder but rather the distance from the centre to the corner of the
38    /// (r,z)-plane.
39    #[serde(deserialize_with = "from_au")]
40    pub radius: f64,
41
42    /// The smallest spatial scale sampled by the code. Structures smaller
43    /// than minScale will not be sampled properly. If one uses spherical
44    /// sampling (see below) this number can also be thought of as the inner
45    /// edge of the grid. This number should not be set smaller than needed,
46    /// because that will cause an undesirably large number of grid points to end up near the
47    /// centre of the model.
48    #[serde(deserialize_with = "from_au")]
49    pub min_scale: f64,
50
51    /// Temperature of the cosmic microwave background in Kelvin.
52    ///
53    /// It defaults to 2.725 K which is the value at zero redshift
54    /// (i.e., the solar neighborhood). One should make sure to set this
55    /// parameter properly when calculating models at a redshift larger
56    /// than zero since
57    /// $$
58    /// T_{\text{CMB}} = T_{\text{CMB},0} (1 + z) \text{K}
59    /// $$
60    /// It should be noted that even though citrus can in this way take
61    /// the change n CMB temperature with increasing $z$ into account,
62    /// it does not (yet) take cosmological effects into account when
63    /// ray tracing (such as stretching of frequencies when using Jansky
64    /// as units).
65    // TODO: Implement cosmological effects in ray tracing
66    pub cmb_temp: f64,
67
68    /// The sinkPoints are grid points that are distributed randomly at
69    /// [`radius`][Parameters::radius] forming the surface of the model. As a photon from within
70    /// the model reaches a sink point it is said to escape and is not tracked
71    /// any longer. The number of sink points is a user-defined quantity since
72    /// the exact number may affect the resulting image as well as the running
73    /// time of the code. One should choose a number that gives a surface
74    /// density large enough not to cause artifacts in the image and low enough
75    /// not to slow down the gridding too much. Since this is model dependent, a
76    /// global best value cannot be given, but a useful range is between a few
77    /// thousands and about ten thousand.
78    pub n_sink_points: usize,
79
80    /// This number is the number of model grid points. The more grid points
81    /// that are used, the longer the code will take to run. Too few points
82    /// however, will cause the model to be under-sampled with the risk of
83    /// getting wrong results. Useful numbers are between a few thousands up to
84    /// about one hundred thousand.
85    pub p_intensity: usize,
86
87    /// If true, line blending is taken into account, however,
88    /// only if there are any overlapping lines among the transitions
89    /// found in the molecular data files. Enabling line blending
90    /// will slow down the calculations considerably, in particular
91    /// if there are multiple molecular data files.
92    ///
93    /// By default, line blending is disabled.
94    pub enable_line_blending: bool,
95
96    /// This parameter specifies the algorithm used by citrus to
97    /// solve the radiative-transfer equations during ray-tracing.
98    /// The default value of zero invokes the algorithm used in
99    /// older versions a value of 1 invokes a new algorithm which is
100    /// much more time-consuming but which produces much smoother
101    /// images, free from step-artifacts.
102    ///
103    /// If none of the four density-linked parameters are provided,
104    /// citrus will attempt to guess the information, in a manner as
105    /// close as possible to the way it was done in version 1.5 and
106    /// earlier. This is safe enough when a single density value is
107    /// returned, and only H2 provided as collision partner in the
108    /// moldata file(s), but more complicated situations can very
109    /// easily result in the code guessing wrongly. For this reason
110    /// we encourage users to make use of these four parameters,
111    /// although in order to preserve backward compatibility with
112    /// old model.c files, we have not (yet) made them mandatory.
113    ///
114    /// ### Note
115    /// There have been additional modifications to the raytracing
116    /// algorithm which have significant effects on the output
117    /// images since citrus-1.5. Image-plane interpolation is now
118    /// employed in areas of the image where the grid point
119    /// spacing is larger than the image pixel spacing. This
120    /// leads both to a smoother image and a shorter processing
121    /// time.
122    pub ray_trace_algorithm: RayTraceAlgorithm,
123
124    /// The sampling algorithm used for the model grid.
125    ///
126    /// By default, the [`UniformExact`][SamplingAlgorithm::UniformExact] is used
127    /// which corresponds to default algorithm in newer versions of citrus.
128    pub sampling_algorithm: SamplingAlgorithm,
129
130    /// If non-zero, citrus performs a direct LTE calculation rather than solving
131    /// for the populations iteratively. This facility is useful for quick checks. The
132    /// default is false, i.e., full non-LTE calculation.
133    pub lte_only: bool,
134
135    /// If true, citrus solves for the level populations as usual, but
136    /// LTE values are used to initialize the populations at the start
137    /// instead of using values at T = 0.
138    pub init_lte: bool,
139
140    /// If true, citrus will calculate the polarized continuum emission.
141    /// This parameter only has effect for continuum images. The resulting cube
142    /// will have three channels containing the Stokes I, Q, and U of the continuum emission
143    /// as according to the theory, the V component is zero. In order for polarization
144    /// to work, a magnetic field must be defined.
145    ///
146    /// When polarization is enabled, the implementation follows the
147    /// `DustPol` code (Padovani et al. 2012), except that the expression
148    /// used in Padovani et al. (2012) given for $\sigma^2$ has shown by
149    /// Adele et al. (2015) to be small by a factor of 2. This correction
150    /// has been implemented in citrus.
151    pub polarization: bool,
152
153    /// This defines the number of solution iterations citrus
154    /// should perform when solving non-LTE level populations.
155    /// The default is currently 17. Note that it is now possible
156    /// to run citrus in an incremental fashion. If the results
157    /// of solving the RTE through `N` iterations are stored in
158    /// a grid file via setting
159    ///  [`Parameters::grid_out_files`],
160    /// then a second run of citrus, reading the grid file
161    /// via [`Parameters::grid_in_file`],
162    /// [`Parameters::n_solve_iters`]` = M>N`, will continue the
163    /// RTE iterations starting at iteration `N`.
164    /// (If you do this, your results will be slightly
165    /// different, in a random way, than if you go to `M`
166    /// iterations in one go, because the random seeds
167    /// will be different.)
168    pub n_solve_iters: usize,
169
170    /// This is the file name of the output file that contains the level
171    /// populations. If this parameter is not set, citrus will not output the
172    /// populations. There is no default value.
173    pub output_file: Option<String>,
174
175    /// This is the file name of the output file that contains the grid,
176    /// populations, and molecular data in binary format. This file is used to
177    /// restart citrus with previously calculated populations. Once the
178    /// populations have been calculated and the binoutputfile has been written,
179    /// citrus can re-raytrace for a different set of image parameters without
180    /// re-calculating the populations. There is no default value.
181    pub binoutput_file: Option<String>,
182
183    /// This is the file name of the output file that contains the grid. If this
184    /// parameter is not set, citrus will not output the grid. The grid file is
185    /// written out as a VTK file. This is a formatted ascii file that can be
186    /// read with a number of 3D visualizing tools (Visualization Tool Kit,
187    /// Paraview, and others). There is no default value.
188    pub grid_file: Option<String>,
189
190    /// Path to the dust opacity file.
191    ///
192    /// This must be prpvided if any continuum images are requested.
193    /// Optional if only line images are requested.
194    ///
195    /// ### File Format
196    /// This table should be a two column ASCII
197    /// file with wavelength in the first column and opacity in the second
198    /// column. Currently citrus uses the same tables as RATRAN from Ossenkopf and
199    /// Henning (1994), and so the wavelength should be given in microns (1e-6
200    /// meters) and the opacity in cm^2/g. This is the only place in citrus where
201    /// SI units are not used. There is
202    /// no default value. A future version of citrus may allow spatial variance
203    /// of the dust opacities, so that opacities can be given as function of x,
204    /// y, and z.
205    pub dust_file: Option<String>,
206    pub grid_in_file: Option<String>,
207
208    /// If this is set non-zero, LIME will use the same random
209    /// number seeds at the start of each solution iteration.
210    /// This has the effect of choosing the same photon directions
211    /// and frequencies for each iteration (although the directions
212    /// and frequencies change randomly from one grid point to
213    /// the next). This has the effect of decoupling any
214    /// oscillation or wandering of the level populations as
215    /// they relax towards convergence from the intrinsic
216    /// Monte Carlo noise of the discrete solution algorithm.
217    /// Best practice might involve alternating episodes
218    /// with `par->resetRNG` =0 and 1, storing the intermediate
219    /// populations via the :ref:`I/O interface <grid-io>`.
220    /// Very little experience has been accumulated as yet
221    /// with this facility.
222    pub reset_rng: bool,
223
224    /// It is now possible to run LIME in two sessions: the
225    /// first to solve the RTE and save the results to file,
226    /// the second to read the file and create raytraced images
227    /// from it. For a session of the first type you should set
228    /// the number of images you specify via the
229    /// :ref:`img <images>` parameter to zero, and give a
230    /// value for one of the elements of
231    /// :ref:`par->gridOutFiles <grid-io>`; for one of
232    /// the second type you set
233    /// :ref:`par->gridInFile <grid-io>` to the name of
234    /// the file you just wrote, and include >0 image
235    /// specifications in :ref:`img <images>`. There is
236    /// a problem however for sessions of the first
237    /// type: if you eventually want full-spectrum cubes
238    /// then you will need some way to tell LIME to solve
239    /// the RTE. In the past LIME has figured out if you
240    /// want this from the presence of spectrum-type images
241    /// in your :ref:`img <images>` list. To replace this
242    /// capability we have added the present parameter.
243    /// Thus, for first-stage sessions (supposing you choose
244    /// to run LIME in that way rather than in the previous
245    /// single-pass style) when you know that you will
246    /// eventually want spectral cubes, you should set
247    /// the present parameter. For all other cases it may
248    /// be ignored.
249    pub do_solve_rte: bool,
250
251    /// We calculate the number density of each of its radiating
252    /// species, at each grid point, by multiplying the abundance
253    /// of the species (returned via the function of that name)
254    /// by a weighted sum of the density values. This parameter
255    /// allows the user to specify the weights in that sum.
256    pub n_mol_weights: Vec<f64>,
257
258    /// Controls how the dust mass density and hence opacity is calculated.
259    /// The calculation of dust mass density in older versions made use of
260    /// a hard-wired average gas density value of 2.4, appropriate
261    /// to a mix of 90% molecular hydrogen and 10% helium. This
262    /// older formula will be used if none of the current four
263    /// parameters are set.
264    pub dust_weights: Vec<f64>,
265
266    pub collisional_partner_mol_weights: Vec<f64>,
267

269    /// Currently recognized values range from 1 to 7 inclusive. E.g if the only colliding
270    /// species of interest in your model is H2, your density function should return a

453impl TryFrom<ParameterInput> for Parameters {
454    type Error = ParamError;
455
456    fn try_from(raw: ParameterInput) -> Result<Self, Self::Error> {
457        if raw.radius <= 0.0 || raw.min_scale <= 0.0 {
458            return Err(ParamError::InvalidDomain);
459        }
460        if raw.min_scale >= raw.radius {
461            return Err(ParamError::MinScaleTooLarge);
462        }
463
464        let domain = Domain {
465            radius: raw.radius,
466            min_scale: raw.min_scale,
467        };
468
469        let outputs = Outputs {
470            populations: raw.output_file,
471            binary: raw.binoutput_file,
472            grid: raw.grid_file,
473            grid_out_files: raw.grid_out_files,
474        };
475
476        let molecular = MolecularConfig {
477            mol_data_files: raw.mol_data_files,
478            g_ir_files: raw.g_ir_data_files,
479            partner_names: raw.collisional_partner_names,
480            partner_ids: raw.collisional_partner_ids,
481            mol_weights: raw.n_mol_weights,
482            partner_weights: raw.collisional_partner_mol_weights,
483        };
484
485        let dust = match raw.dust_file {
486            Some(path) => DustConfig::Table {
487                path,
488                weights: raw.dust_weights,
489            },
490            None => DustConfig::None,
491        };
492
493        let solver = SolverConfig {
494            n_iters: raw.n_solve_iters,
495            lte_only: raw.lte_only,
496            init_lte: raw.init_lte,
497            solve_rte: raw.do_solve_rte,
498            reset_rng: raw.reset_rng,
499        };
500
501        let grid = GridConfig {
502            n_points: raw.p_intensity,
503            n_sink_points: raw.n_sink_points,
504            sampling: raw.sampling_algorithm,
505        };
506
507        let mode = match raw.grid_in_file {
508            Some(file) if !file.is_empty() => RunMode::Restart(RestartRun { grid_in_file: file }),
509            _ => RunMode::Fresh(FreshRun { grid_in_file: None }),
510        };
511
512        Ok(Self {
513            domain,
514            cmb_temp: raw.cmb_temp,
515            grid,
516            solver,
517            outputs,
518            molecular,
519            dust,
520            raytrace: raw.ray_trace_algorithm,
521            polarization: raw.polarization,
522            line_blending: raw.enable_line_blending,
523            mode,
524        })
525    }
526}

528/// Sampling algorithms to generate grid points for modeling.
529#[derive(Deserialize, Debug, Default)]
530pub enum SamplingAlgorithm {
531    /// This algorithm can be used uniform sampling in Log(radius)
532    /// which is useful for models with a central condensation
533    /// (i.e., envelopes, disks)
534    Uniform,
535
536    /// This algorithm generates grid points with exact
537    /// spherical rotation symmetry.
538    #[default]
539    UniformExact,
540
541    /// This generates grid points with a uniform-biased
542    /// sampling in cartesian coordinates.
543    ///
544    /// This is useful for models with no central condensation
545    /// (molecular clouds, galaxies, slab geometries, etc.)
546    UniformBiased,
547
548    /// This is a newer algorithm which can quickly generate points with a
549    /// distribution which accurately follows any feasible density function.
550    /// including with sharp step-changes.
551    ///
552    /// The contained value is a vector of [`density maxima`][GridDensityMaxima] which
553    /// the user can provide to help guide the point distribution.
554    ///
555    /// This algorithm also incorporates a quasi-random choice of
556    /// point candidates which avoids the requirement for the relatively
557    /// time-consuming post-gridding smoothing phase. Therefore,
558    /// supplying your own density function will give full control
559    /// over the distribution of points. Currently, [`qhull`] is
560    /// being used to triangulate the point distribution.
561    // TODO: Think about getting rid of qhull dependency and implementing a pure Rust solution for Delaunay triangulation.
562    Modern(Vec<GridDensityMaxima>),
563}
564
565impl SamplingAlgorithm {
566    #[must_use]
567    pub fn is_legacy(&self) -> bool {
568        !matches!(self, SamplingAlgorithm::Modern(_))
569    }
570}

711impl TryFrom<ImageInput> for ImageKind {
712    type Error = ImageError;

714    fn try_from(raw: ImageInput) -> Result<Self, Self::Error> {
715        let image = Image {
716            spatial: Spatial {
717                pxls: raw.pxls,
718                img_res: raw.img_res,
719                distance: raw.distance,
720            },
721            orientation: Orientation {
722                theta: raw.theta,
723                phi: raw.phi,
724                inclination: raw.inclination,
725                position_angle: raw.position_angle,
726                azimuth: raw.azimuth,
727                // rotation_matrix: raw.rotation_matrix,
728            },
729            filename: raw.filename,
730            unit: raw.unit,
731            source_velocity: raw.source_velocity,
732            interpolate_vels: raw.do_interpolate_vels,
733            pixels: raw.pixel,
734        };
735
736        if raw.do_line {
737            let freq = if let Some(trans) = raw.trans {
738                LineFreq::Transition {
739                    trans,
740                    mol_i: raw.mol_i,
741                }
742            } else {
743                LineFreq::Explicit(raw.freq)
744            };
745
746            let spectral = match (raw.bandwidth, raw.vel_res, raw.nchan) {
747                (b, v, 0) if b > 0.0 && v > 0.0 => SpectralAxis::BandwidthVelRes {
748                    bandwidth: b,
749                    vel_res: v,
750                },
751
752                (b, 0.0, n) if b > 0.0 && n > 0 => SpectralAxis::BandwidthNchan {
753                    bandwidth: b,
754                    nchan: n,
755                },
756
757                (0.0, v, n) if v > 0.0 && n > 0 => SpectralAxis::VelResNchan {
758                    vel_res: v,
759                    nchan: n,
760                },
761
762                _ => return Err(ImageError::InvalidSpectralDefinition),
763            };
764
765            Ok(ImageKind::Line(LineImage {
766                image,
767                freq,
768                spectral,
769            }))
770        } else {
771            if raw.freq <= 0.0 {
772                return Err(ImageError::ContinuumNeedsFreq);
773            }
774
775            Ok(ImageKind::Continuum(ContinuumImage {
776                image,
777                freq: raw.freq,
778            }))
779        }
780    }
781}

799impl Config {
800    /// Load the configuration from a file
801    ///
802    /// # Errors
803    /// This function may fail if the file cannot be read or if the configuration is invalid.
804    pub fn from_path(path: &str) -> Result<Self> {
805        let content = fs::read_to_string(path)?; // Propagate file read errors
806        let config: Config = toml::from_str(&content)?; // Propagate config parsing errors
807        Ok(config)
808    }
809
810    /// Parse the configuration for correctness and consistency
811    ///
812    /// # Errors
813    /// This function may fail if the configuration is invalid.
814    ///
815    /// # Panics
816    /// This function may panic if the model is invalid.
817    pub fn parse<M: Model>(&mut self, model: &M) -> Result<()> {
818        let mut r = Vec3::zero();
819
820        self.parameters.ncell = self.parameters.p_intensity + self.parameters.n_sink_points;
821        self.parameters.radius_squ = self.parameters.radius * self.parameters.radius;
822        self.parameters.min_scale_squ = self.parameters.min_scale * self.parameters.min_scale;
823        self.parameters.n_solve_iters_done = 0;
824        self.parameters.use_abun = true;
825
826        if self.parameters.grid_in_file.is_none() {
827            assert!(
828                self.parameters.radius > 0.0,
829                "Radius must be positive and non-zero."
830            );
831            assert!(
832                self.parameters.min_scale > 0.0,
833                "Minimum scale must be positive and non-zero."
834            );
835            assert!(
836                self.parameters.p_intensity > 0,
837                "Number of intensity points must be positive and non-zero."
838            );
839            assert!(
840                self.parameters.n_sink_points > 0,
841                "Number of sink points must be positive and non-zero."
842            );
843        }
844
845        self.parameters.grid_density_global_max = 1.0;
846        self.parameters.n_densities = 0;
847
848        if self.parameters.grid_in_file.is_none() {
849            // Read the grid file in FITS format
850            // TODO: Currently not implemented
851            todo!()
852        }
853
854        if self.parameters.n_densities == 0 {
855            // This means the FITS file read above did not return any density values,
856            // In this case the default density function will not suffice.
857            // It is not straightforward to check if the user type that will implement
858            // the [`Model`] trait is using the default density function or did they
859            // override it with their own due to lack of runtime reflection in Rust.
860            // Therefore for now my plan is to leave this part as a TODO.
861            todo!()
862        }
863
864        // Not sure this is needed?
865        self.parameters.grid_density_global_max = 1.0;
866
867        let mut temp_point_density = model.grid_density(&self.parameters, &r);
868
869        if temp_point_density.is_infinite() || temp_point_density.is_nan() {
870            eprintln!("There is a singularity in the grid density function at the origin.");
871        } else if temp_point_density <= 0.0 {
872            eprintln!("The grid density function is zero at the origin.");
873        } else if temp_point_density > self.parameters.grid_density_global_max {
874            self.parameters.grid_density_global_max = temp_point_density;
875        }
876
877        if temp_point_density.is_infinite()
878            || temp_point_density.is_nan()
879            || temp_point_density <= 0.0
880        {
881            r = self.parameters.min_scale * Vec3::one();
882            temp_point_density = model.grid_density(&self.parameters, &r);
883
884            if !temp_point_density.is_infinite()
885                && !temp_point_density.is_nan()
886                && temp_point_density > 0.0
887            {
888                self.parameters.grid_density_global_max = temp_point_density;
889            } else {
890                // Hmm ok, let's try a spread of random locations
891                let mut rand_gen = if true {
892                    // Use fixed seed for reproducibility
893                    // Note: SeedableRng::seed_from_u64 takes a u64 seed
894                    StdRng::seed_from_u64(140_978)
895                } else {
896                    let mut thread_rng = rand::rng();
897                    // Seed from the system's entropy source for non-reproducible randomness
898                    // StdRng::from_entropy is a good way to get a random seed
899                    StdRng::from_rng(&mut thread_rng)
900                };
901                println!("Random number generator initialized.");
902                let mut found_good_value = false;
903                for _ in 0..100 {
904                    // TODO: Better way to do this?
905                    r.x = rand_gen.random_range(-self.parameters.radius..self.parameters.radius);
906                    r.y = rand_gen.random_range(-self.parameters.radius..self.parameters.radius);
907                    r.z = rand_gen.random_range(-self.parameters.radius..self.parameters.radius);
908
909                    temp_point_density = model.grid_density(&self.parameters, &r);
910                    if !temp_point_density.is_infinite()
911                        && !temp_point_density.is_nan()
912                        && temp_point_density > 0.0
913                    {
914                        found_good_value = true;
915                        break;
916                    }
917                }
918
919                if found_good_value {
920                    if temp_point_density > self.parameters.grid_density_global_max {
921                        self.parameters.grid_density_global_max = temp_point_density;
922                    }
923                } else if let SamplingAlgorithm::Modern(grid_density_maxima) =
924                    &self.parameters.sampling_algorithm
925                {
926                    assert!(
927                        !grid_density_maxima.is_empty(),
928                        "Could not find a non-pathological grid density value, and no grid density maxima were provided to guide the sampling algorithm."
929                    );
930
931                    for m in grid_density_maxima {
932                        if m.value > self.parameters.grid_density_global_max {
933                            self.parameters.grid_density_global_max = m.value;
934                        }
935                    }
936                } else {
937                    panic!(
938                        "Grid density maxima only supported for Modern sampling algorithm. Could not find a non-pathological grid density value."
939                    );
940                }
941            }
942        }
943
944        for i in 0..constants::NUM_OF_GRID_STAGES {
945            if !self.parameters.grid_out_files[i].is_empty() {
946                self.parameters.write_grid_at_stage[i] = true;
947            }
948        }
949
950        /*
951        Now we need to calculate the cutoff value used in `calc_source_fn()`. The issue is
952        to decide between
953
954          y = (1 - exp[-x])/x
955
956        or the approximation using the Taylor expansion of exp(-x), which to 3rd order
957        is
958
959          y ~ 1 - x/2 + x^2/6.
960
961        The cutoff will be the value of abs(x) for which the error in the exact
962        expression equals the next unused Taylor term, which is x^3/24. This error can
963        be shown to be given for small |x| by epsilon/|x|, where epsilon is the
964        floating-point precision of the computer. Hence the cutoff evaluates to
965
966          |x|_cutoff = (24*epsilon)^{1/4}.
967
968        */
969        self.parameters.taylor_cutoff = (24.0 * f64::EPSILON).powf(0.25);
970
971        // Allocate pixel space and parse image information
972        for (i, image) in self.images.iter_mut().enumerate() {
973            if image.nchan == 0 && image.vel_res < 0.0 {
974                // User has set neither `nchan` nor `vel_res`
975                // One of the two are required for a line image
976                // Therefore, we assume continuum image
977                if self.parameters.polarization {
978                    image.nchan = 3;
979                } else {
980                    image.nchan = 1;
981                }
982                assert!(
983                    image.freq != 0.0,
984                    "You must set a frequency for continuum image."
985                );
986                if image.trans.is_some() || image.bandwidth > 0.0 {
987                    eprintln!(
988                        "Transition number and bandwidth are not relevant for a continuum image. They will be ignored."
989                    );
990                    image.do_line = false;
991                }
992            } else {
993                // User has set either `nchan` or `vel_res`
994                // Therefore, we assume line image
995
996                /*
997                For a valid line image, the user must set one of the following pairs:
998                bandwidth, velres (if they also set nchan, this is overwritten)
999                bandwidth, nchan (if they also set velres, this is overwritten)
1000                velres, nchan (if they also set bandwidth, this is overwritten)
1001
1002                The presence of one of these combinations at least is checked here, although the
1003                actual calculation is done in raytrace(), because it depends on moldata info
1004                which we have not yet got.
1005                */
1006                if image.bandwidth > 0.0 && image.vel_res > 0.0 {
1007                    if image.nchan > 0 {
1008                        eprintln!(
1009                            "As image had both bandwidth and velres set, nchan will be overwritten."
1010                        );
1011                    }
1012                } else if image.nchan == 0 || image.bandwidth <= 0.0 && image.vel_res <= 0.0 {
1013                    panic!("You must set either nchan, bandwidth, or velres for a line image.");
1014                }
1015                // Check that we have keywords which allow us to calculate the image
1016                // frequency (if necessary) after reading in the moldata file:
1017                if image.trans.is_some() {
1018                    // User has set of `trans`, posssibly also `freq`
1019                    if image.freq > 0.0 {
1020                        eprintln!(
1021                            "Ignoring user-set frequency since transition number is set. The frequency will be calculated from the molecular data file."
1022                        );
1023                    }
1024                    if self.parameters.n_species > 1 {
1025                        let msg = format!(
1026                            "WARNING: Image {i} did not have ``mol_i`` set, \
1027                                Therefore, first molecule will be used.",
1028                        );
1029                        eprintln!("{msg}");
1030                        image.mol_i = 0;
1031                    }
1032                } else if image.freq < 0.0 {
1033                    // User has not set `trans`, nor `freq`
1034                    bail!(
1035                        "You must either set `trans` or `freq` for a line image (and optionally the `mol_i`"
1036                    );
1037                } // else user has set `freq`
1038                image.do_line = true;
1039            } // End of check for line or continuum image
1040            if image.img_res < 0.0 {
1041                bail!("You must set image resolution.");
1042            }
1043            if image.pxls == 0 {
1044                bail!("You must set number of pixels.");
1045            }
1046            if image.distance <= 0.0 {
1047                bail!("You must set distance to source.");
1048            }
1049            image.img_res = (image.img_res / 3600.0).to_radians();
1050            image.pixel = {
1051                let mut v = Vec::with_capacity(image.pxls * image.pxls);
1052                for _ in 0..(image.pxls * image.pxls) {
1053                    v.push(Spec::default());
1054                }
1055                v
1056            };
1057            for spec in &mut image.pixel {
1058                spec.intense = RVector::zeros(image.nchan);
1059                spec.tau = RVector::zeros(image.nchan);
1060            }
1061
1062            // Calculate the rotation matrix
1063            /*
1064            The image rotation matrix is used within traceray() to transform the coordinates
1065            of a vector (actually two vectors - the ray direction and its starting point) as
1066            initially specified in the observer-anchored frame into the coordinate frame of
1067            the model. In linear algebra terms, the model-frame vector v_mod is related to
1068            the vector v_obs as expressed in observer- (or image-) frame coordinates via the
1069            image rotation matrix R by
1070
1071                v_mod = R * v_obs,				1
1072
1073            the multiplication being the usual matrix-vector style. Note that the ith row of
1074            R is the ith axis of the model frame with coordinate values expressed in terms
1075            of the observer frame.
1076
1077            The matrix R can be broken into a sequence of several (3 at least are needed for
1078            full degrees of freedom) simpler rotations. Since these constituent rotations
1079            are usually easier to conceive in terms of rotations of the model in the
1080            observer framework, it is convenient to invert equation (1) to give
1081
1082                v_obs = R^T * v_mod,			2
1083
1084            where ^T here denotes transpose. Supposing now we rotate the model in a sequence
1085            R_3^T followed by R_2^T followed by R_1^T, equation (2) can be expanded to give
1086
1087                v_obs = R_1^T * R_2^T * R_3^T * v_mod.	3
1088
1089            Inverting everything to return to the format of equation (1), which is what we
1090            need, we find
1091
1092                v_mod = R_3 * R_2 * R_1 * v_obs.		4
1093
1094            citrus provides two different schemes of {R_1, R_2, R_3}: {PA, phi, theta} and
1095            {PA, inclination, azimuth}. As an example, consider phi, which is a rotation of
1096            the model from the observer Z axis towards the X. The matching obs->mod rotation
1097            matrix is therefore
1098
1099                        ( cos(ph)  0  -sin(ph) )
1100                        (                      )
1101                R_phi = (    0     0     1     ).
1102                        (                      )
1103                        ( sin(ph)  0   cos(ph) )
1104
1105                */
1106            let do_theta_phi = image.inclination < -900.0
1107                || image.azimuth < -900.0
1108                || image.position_angle < -900.0;
1109            if do_theta_phi {
1110                // For the present position angle is not implemented
1111                // for the theta/phi scheme, so we will just load the
1112                // the identity matrix
1113                image.rotation_matrix = RMatrix::eye(3);
1114            } else {
1115                // Load position angle matrix
1116                let cos_pa = image.position_angle.cos();
1117                let sin_pa = image.position_angle.sin();
1118                image.rotation_matrix = array![
1119                    [cos_pa, -sin_pa, 0.0],
1120                    [sin_pa, cos_pa, 0.0],
1121                    [0.0, 0.0, 1.0]
1122                ];
1123            }
1124            let aux_rotation_matrix = if do_theta_phi {
1125                // Load phi rotation matrix R_phi
1126                let cos_phi = image.phi.cos();
1127                let sin_phi = image.phi.sin();
1128                array![
1129                    [cos_phi, 0.0, -sin_phi],
1130                    [0.0, 1.0, 0.0],
1131                    [sin_phi, 0.0, cos_phi]
1132                ]
1133            } else {
1134                // Load inclination matrix R_incl
1135                let cos_incl = (image.inclination + std::f64::consts::PI).cos();
1136                let sin_incl = (image.inclination + std::f64::consts::PI).sin();
1137                array![
1138                    [cos_incl, 0.0, -sin_incl],
1139                    [0.0, 1.0, 0.0],
1140                    [sin_incl, 0.0, cos_incl]
1141                ]
1142            };
1143            // Multiply the two matrices
1144            let _temp_matrix = image.rotation_matrix.dot(&aux_rotation_matrix);
1145        }
1146
1147        self.parameters.n_line_images = 0;
1148        self.parameters.n_cont_images = 0;
1149        self.parameters.do_interpolate_vels = false;
1150
1151        for image in &mut self.images {
1152            if image.do_line {
1153                self.parameters.n_line_images += 1;
1154            } else {
1155                self.parameters.n_cont_images += 1;
1156            }
1157            if image.do_interpolate_vels {
1158                self.parameters.do_interpolate_vels = true;
1159            }
1160        }
1161
1162        if self.parameters.n_cont_images > 0 {
1163            match &self.parameters.dust_file {
1164                Some(dust) if !dust.is_empty() => {
1165                    // Open the dust file and check if it exists and if it is empty
1166                    let path = Path::new(dust);
1167                    if path.exists() {
1168                        match fs::metadata(path) {
1169                            Ok(metadata) => {
1170                                if metadata.len() == 0 {
1171                                    eprintln!("File {dust} is empty.");
1172                                }
1173                            }
1174                            Err(err) => {
1175                                eprintln!("Error accessing file {dust}: {err}");
1176                            }
1177                        }
1178                    }
1179                }
1180                _ => {
1181                    bail!("You must set dust parameters for continuum images.");
1182                }
1183            }
1184        }
1185
1186        self.parameters.use_vel_func_in_raytrace = self.parameters.n_line_images > 0
1187            && self.parameters.ray_trace_algorithm == RayTraceAlgorithm::Legacy
1188            && !self.parameters.do_pre_grid;
1189
1190        self.parameters.edge_vels_available = false;
1191
1192        if self.parameters.lte_only {
1193            if self.parameters.n_solve_iters > 0 {
1194                let msg = "Requesting `n_solve_iters > 0` in LTE only mode \
1195            will have no effect";
1196                eprintln!("{msg}");
1197            } else if self.parameters.n_solve_iters <= self.parameters.n_solve_iters_done {
1198                let msg = "Requesting `n_solve_iters <= n_solve_iters_done` in LTE only mode \
1199            will have no effect";
1200                eprintln!("{msg}");
1201            }
1202        }
1203
1204        let _mol_data: Option<MolData> = if self.parameters.n_species > 0 {
1205            // defaults::mol_data(self.parameters.n_species)
1206            todo!()
1207        } else {
1208            None
1209        };
1210
1211        let _density_power = if self.parameters.sampling_algorithm.is_legacy() {
1212            constants::DENSITY_EXP
1213        } else {
1214            constants::TREE_EXP
1215        };
1216
1217        Ok(())
1218    }
1219}
