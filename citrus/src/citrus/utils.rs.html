14/// The source function `S` is defined as `j_nu/alpha`, which is clearly not
15/// defined for `alpha==0`. However `S` is used in the algorithm only in the
16/// term `(1-exp[-alpha*ds])*S`, which is defined for all values of alpha.
17/// The present function calculates this term and returns it in the
18/// argument remnantSnu. For values of `abs(alpha*ds)` less than a pre-
19/// calculated cutoff supplied in `ConfigInfo`, a Taylor approximation is
20/// used.
21/// Note that the same cutoff condition holds for replacement of
22/// `exp(-dTau)` by its Taylor expansion to 3rd order.
23#[must_use]
24pub fn calc_source_fn(dtau: f64, taylor_cutoff: f64) -> (f64, f64) {
25    let remant_snu: f64;
26    let exp_dtau: f64;
27    if dtau.abs() < taylor_cutoff {
28        remant_snu = 1.0 - dtau * (1.0 - dtau * (1.0 / 3.0)) * (1.0 / 2.0);
29        exp_dtau = 1.0 - dtau * (remant_snu);
30    } else {
31        exp_dtau = (-dtau).exp();
32        remant_snu = (1.0 - exp_dtau) / dtau;
33    }
34
35    (remant_snu, exp_dtau)
36}

38#[must_use]
39pub fn planck_fn(freq: f64, t_kelvin: f64) -> f64 {
40    if t_kelvin < constants::CITRUS_GLOBAL_EPS {
41        0.0
42    } else {
43        let wn = freq / cc::SPEED_OF_LIGHT_IN_VACUUM_SI;
44        if cc::PLANCK_CONSTANT_SI * freq > 100.0 * cc::BOLTZMANN_CONSTANT_SI * t_kelvin {
45            2.0 * cc::PLANCK_CONSTANT_SI
46                * wn
47                * wn
48                * freq
49                * (-cc::PLANCK_CONSTANT_SI * freq / cc::BOLTZMANN_CONSTANT_SI / t_kelvin).exp()
50        } else {
51            2.0 * cc::PLANCK_CONSTANT_SI * wn * wn * freq
52                / ((cc::PLANCK_CONSTANT_SI * freq / cc::BOLTZMANN_CONSTANT_SI / t_kelvin - 1.0)
53                    .exp()
54                    - 1.0)
55        }
56    }
57}

65pub fn get_erf(x0: f64, x1: f64) -> f64 {
66    let val0 = erf(x0);
67    let val1 = erf(x1);
68    let diff = val1 - val0;
69    let delta_x = x1 - x0;
70    (diff / delta_x).abs()
71}

73/// This function:
74/// 1. Converts frequency to wavelength in log10 space
75/// 2. Handles extrapolation outside the table range
76/// 3. Converts results from cm²/g to m²/kg (0.1 factor)
77///
78/// Parameters:
79/// - freq: Frequency
80/// - lamtab: Array of log10 wavelength values
81/// - kaptab: Array of log10 opacity values
82/// - n_entries: Number of entries in the arrays
83pub fn interpolate_kappa(freq: f64, lam: &RVecView, kap: &RVecView) -> Result<f64, SplineError> {
84    if lam.len() != kap.len() {
85        return Err(SplineError::DifferentLengths);
86    }
87
88    // Create the spline
89    let spline = CubicSpline::new(lam, kap)?;
90
91    // Calculate log10 of wavelength (SPEED_OF_LIGHT_SI / freq)
92    let loglam = (cc::SPEED_OF_LIGHT_IN_VACUUM_SI / freq).log10();
93
94    let n_entries = lam.len();
95    let kappa_log10: f64;
96
97    if loglam < lam[0] {
98        // Below range: linear extrapolation in log space
99        kappa_log10 = kap[0] + (loglam - lam[0]) * (kap[1] - kap[0]) / (lam[1] - lam[0]);
100    } else if loglam > lam[n_entries - 1] {
101        // Above range: linear extrapolation in log space
102        kappa_log10 = kap[n_entries - 2]
103            + (loglam - lam[n_entries - 2]) * (kap[n_entries - 1] - kap[n_entries - 2])
104                / (lam[n_entries - 1] - lam[n_entries - 2]);
105    } else {
106        // In range: use the spline
107        kappa_log10 = spline.interpolate(loglam)?;
108    }
109
110    // Convert from log10 opacity and apply unit conversion factor (cm²/g to m²/kg)
111    let kappa = 0.1 * 10f64.powf(kappa_log10);
112
113    Ok(kappa)
114}

127    if par.collisional_partner_user_set_flags == 0 {
128        cc::ATOMIC_UNIT_OF_MASS_SI * 2.4 * dens[0] / gtd
129    } else {
130        let mut gas_mass_density_amus = 0.0;
131        for i in 0..par.n_densities {
132            gas_mass_density_amus += dens[i] * par.collisional_partner_mol_weights[i];
133        }
134        cc::ATOMIC_UNIT_OF_MASS_SI * gas_mass_density_amus / gtd

173    pub struct CubicSpline {
174        x: RVector,
175        y: RVector,
176        // Second derivatives at each point
177        y2: RVector,
178    }
179
180    impl CubicSpline {
181        /// Create a new cubic spline interpolator
182        pub fn new(x: &RVecView, y: &RVecView) -> Result<Self, SplineError> {
183            if x.len() != y.len() {
184                return Err(SplineError::DifferentLengths);
185            }
186            if x.len() < 2 {
187                return Err(SplineError::InsufficientPoints);
188            }
189
190            // Check that x is sorted in ascending order
191            for i in 1..x.len() {
192                if x[i] <= x[i - 1] {
193                    return Err(SplineError::UnsortedPoints);
194                }
195            }
196
197            let n = x.len();
198            let mut y2 = RVector::zeros(n);
199
200            // This computes the second derivatives needed for the spline
201            // Natural spline conditions: y''(x[0]) = y''(x[n-1]) = 0
202            let mut u = vec![0.0; n - 1];
203
204            // Decomposition loop of the tridiagonal algorithm
205            for i in 1..n - 1 {
206                let sig = (x[i] - x[i - 1]) / (x[i + 1] - x[i - 1]);
207                let p = sig * y2[i - 1] + 2.0;
208                // Check for numerical stability
209                if p.abs() < f64::EPSILON {
210                    return Err(SplineError::NumericalError);
211                }
212                y2[i] = (sig - 1.0) / p;
213                u[i] =
214                    (y[i + 1] - y[i]) / (x[i + 1] - x[i]) - (y[i] - y[i - 1]) / (x[i] - x[i - 1]);
215                u[i] = (6.0 * u[i] / (x[i + 1] - x[i - 1]) - sig * u[i - 1]) / p;
216            }
217
218            // Back-substitution loop of the tridiagonal algorithm
219            for i in (0..n - 1).rev() {
220                y2[i] = y2[i] * y2[i + 1] + u[i];
221            }
222
223            Ok(CubicSpline {
224                x: x.to_owned(),
225                y: y.to_owned(),
226                y2,
227            })
228        }
229
230        /// Interpolate at the given x value
231        pub fn interpolate(&self, x: f64) -> Result<f64, SplineError> {
232            // Find the right interval using binary search
233            let n = self.x.len();
234
235            // Handle out-of-bounds
236            if x < self.x[0] || x > self.x[n - 1] {
237                return Err(SplineError::OutOfBounds);
238            }
239
240            // Binary search to find the right interval
241            let mut low = 0;
242            let mut high = n - 1;
243
244            while high - low > 1 {
245                let mid = usize::midpoint(high, low);
246                if self.x[mid] > x {
247                    high = mid;
248                } else {
249                    low = mid;
250                }
251            }
252
253            // Now x is between self.x[low] and self.x[high]
254            let h = self.x[high] - self.x[low];
255            if h == 0.0 {
256                return Err(SplineError::PointsTooClose);
257            }
258
259            // Cubic spline polynomial evaluation
260            let a = (self.x[high] - x) / h;
261            let b = (x - self.x[low]) / h;
262
263            // Evaluate the spline polynomial
264            let y = a * self.y[low]
265                + b * self.y[high]
266                + ((a * a * a - a) * self.y2[low] + (b * b * b - b) * self.y2[high]) * (h * h)
267                    / 6.0;
268
269            Ok(y)
270        }
271
272        /// Evaluate the spline with extrapolation outside bounds
273        /// This is a special method to handle cases where x is outside the range
274        pub fn eval_with_extrapolation(&self, x: f64) -> Result<f64, SplineError> {
275            let n = self.x.len();
276
277            // Handle in-bounds case
278            if x >= self.x[0] && x <= self.x[n - 1] {
279                return self.interpolate(x);
280            }
281
282            // If we're out of bounds, use linear extrapolation
283            if x < self.x[0] {
284                // Linear extrapolation below range using the first two points
285                let slope = (self.y[1] - self.y[0]) / (self.x[1] - self.x[0]);
286                Ok(self.y[0] + slope * (x - self.x[0]))
287            } else {
288                // Linear extrapolation above range using the last two points
289                let slope = (self.y[n - 1] - self.y[n - 2]) / (self.x[n - 1] - self.x[n - 2]);
290                Ok(self.y[n - 1] + slope * (x - self.x[n - 1]))
291            }
292        }
293    }
294    #[derive(Debug, Clone, PartialEq)]
295    pub enum SplineError {
296        /// Input arrays have different lengths
297        DifferentLengths,
298        /// Not enough points provided for interpolation
299        InsufficientPoints,
300        /// Input x values must be in strictly ascending order
301        UnsortedPoints,
302        /// Interpolation requested outside range of input data
303        OutOfBounds,
304        /// Points in the input data are too close together
305        PointsTooClose,
306        /// Numerical instability detected during calculation
307        NumericalError,
308    }
309
310    impl std::fmt::Display for SplineError {
311        fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
312            match self {
313                SplineError::DifferentLengths => {
314                    write!(f, "Input arrays must have the same length")
315                }
316                SplineError::InsufficientPoints => {
317                    write!(f, "Need at least two points for interpolation")
318                }
319                SplineError::UnsortedPoints => {
320                    write!(f, "Input x values must be in strictly ascending order")
321                }
322                SplineError::OutOfBounds => write!(f, "Interpolation x value out of bounds"),
323                SplineError::PointsTooClose => write!(f, "Interpolation points too close"),
324                SplineError::NumericalError => write!(f, "Numerical error during computation"),
325            }
326        }
327    }

340    //! A simple quasi-random number generator based on the Halton sequence.
341    //! The Halton sequence is a sequence of numbers that are generated by
342    //! taking the fractional part of a number raised to a power. The Halton
343    //! sequence is defined for any positive integer `n`, and the power is
344    //! defined as `2^n`.
345
346    /// Halton State
347    pub struct Halton {
348        index: usize,
349        dimension: usize,
350    }
351
352    #[derive(Debug)]
353    pub enum HaltonError {
354        DimensionTooLarge,
355    }
356
357    impl std::fmt::Display for HaltonError {
358        fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
359            match self {
360                HaltonError::DimensionTooLarge => {
361                    write!(f, "Halton sequence cannot have dimension greater than 1229")
362                }
363            }
364        }
365    }
366
367    impl std::error::Error for HaltonError {}
368
369    impl Halton {
370        pub fn new(dimension: usize) -> Result<Self, HaltonError> {
371            if dimension == 0 || dimension > PRIMES.len() {
372                return Err(HaltonError::DimensionTooLarge);
373            }
374            Ok(Halton {
375                index: 0,
376                dimension,
377            })
378        }
379
380        pub fn next_point(&mut self) -> Vec<f64> {
381            let mut point = Vec::with_capacity(self.dimension);
382            for &base in &PRIMES[..self.dimension] {
383                point.push(halton(self.index, base));
384            }
385            self.index += 1;
386            point
387        }
388    }
389
390    fn halton(mut index: usize, base: usize) -> f64 {
391        let mut result = 0.0;
392        let mut f = 1.0 / base as f64;
393        while index > 0 {
394            result += f * (index % base) as f64;
395            index /= base;
396            f /= base as f64;
397        }
398        result
399    }
