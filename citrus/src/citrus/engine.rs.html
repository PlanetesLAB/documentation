18/// Run the code
19///
20/// # Errors
21/// This function may fail if the configuration is invalid.
22///
23/// # Panics
24/// This function may panic if the model is invalid.
25pub fn run(
26    par: &mut ParameterInput,
27    imgs: &mut [ImageInput],
28    mol_data: Option<&Vec<MolData>>,
29) -> Result<()> {
30    if par.n_solve_iters > par.n_solve_iters_done || par.lte_only {
31        par.do_solve_rte = true;
32    }
33    par.do_mol_calcs = par.do_solve_rte || par.n_line_images > 0;
34    if par.do_mol_calcs && par.mol_data_files.is_empty() {
35        bail!("You must set the molecular data file.");
36    }
37
38    if !par.do_mol_calcs && par.init_lte {
39        let msg = "WARNING: Your choice of `init_lte` will have no effect \
40        as no molecular calculations are requested.";
41        eprintln!("{msg}");
42    }
43
44    if par.n_species > 0 && !par.do_mol_calcs {
45        bail!(
46            "If you only want to do continuum calculations, \
47        you must supply no molecular data files."
48        );
49    }
50
51    let mut gp = {
52        check_user_density_weights(par)?;
53        grid::read_or_build_grid(par)?
54    };
55
56    let lam_kap = match &par.dust_file {
57        Some(dust) if !dust.is_empty() => Some(read_dust_file(dust)?),
58        _ => {
59            eprintln!("No dust file provided.");
60            None
61        }
62    };
63
64    let mol_slice = mol_data.as_ref().expect("mol_data is None").as_slice();
65
66    if par.n_cont_images > 0 {
67        for img in imgs.iter_mut() {
68            raytrace(img, gp.as_mut_slice(), par, mol_slice, lam_kap.as_ref())?;
69        }
70    }
71
72    Ok(())
73}
